 
#!/bin/bash
# Global pre-commit hook to check for Python installation
# Check for Python
if command -v python3 > /dev/null 2>&1; then
    echo "Python3 is available. Commit review feature is enabled."
elif command -v python > /dev/null 2>&1; then
    echo "Python is available. Commit review feature is enabled."
else
    echo "WARNING: Python3 is not installed. Commit review functionality will not work." >&2
fi
# Function to scan for sensitive information
scan_for_secrets() {
    local file=$1
    local secret_found=0
    # Define patterns to detect secrets (add more as needed)
    patterns=(
        "sk-[a-zA-Z0-9\_-]{36,}"                # OpenAI API Key
        "AKIA[0-9A-Z]{16}"                  # AWS Access Key ID
        "ASIA[0-9A-Z]{16}"                  # AWS Temporary Access Key
        "AIza[0-9A-Za-z_-]{35}"             # Google API Key
        "AIzaSy[A-Za-z0-9_-]{35}"           # Google OAuth Key
        "[0-9a-fA-F]{32}"                   # Generic 32-character key (e.g., MD5 hash)
        "[0-9a-fA-F]{40}"                   # Generic 40-character key (e.g., SHA1 hash)
        "[A-Za-z0-9_-]{64}"                 # Generic 64-character key (e.g., JWT)
        "xox[baprs]-[0-9A-Za-z]{10,48}"     # Slack Token
        "[A-Za-z0-9]{20}-us[0-9]{2}"        # MailChimp API Key
        "SG\.[A-Za-z0-9_-]{22}\.[A-Za-z0-9_-]{43}" # SendGrid API Key
        "sq0atp-[0-9A-Za-z\\-_]{22}"        # Square Access Token
        "sq0csp-[0-9A-Za-z\\-_]{43}"        # Square Client Secret
        "sk_live_[0-9a-zA-Z]{24}"           # Stripe Live Secret Key
        "[a-zA-Z0-9_-]{39}"                 # GitHub Personal Access Token
        "pt_[a-zA-Z0-9]{20}"                # PayPal Token
        "(password\s*=\s*['\"].+['\"]?)"    # Password patterns
        "(PRIVATE_KEY.*?-----BEGIN.*?-----END)" # RSA Private Key
    )
    for pattern in "${patterns[@]}"; do
        if grep -Eq "$pattern" "$file"; then  # -E for extended regex
            echo -e "\e[31mSensitive information detected in $file\e[0m"
            echo -e "\e[31mPattern: $pattern\e[0m"
            secret_found=1
        fi
    done
    
    return $secret_found
}
# Function to check for disallowed file types
check_disallowed_files() {
    local file=$1
    local disallowed_found=0
    # Define disallowed file extensions
    disallowed_extensions=(
        ".crt"         # Certificate files
        ".cer"
        ".ca-bundle"
        ".p7b"
        ".p7c"
        ".p7s"
        ".pem"
        ".jceks"
        ".key"         # Keystore files
        ".keystore"
        ".jks"
        ".p12"         # Combined certificate and key files
        ".pfx"
    )
    for ext in "${disallowed_extensions[@]}"; do
        if [[ "$file" == *"$ext" ]]; then
            echo -e "\e[31mDisallowed file detected: $file (extension: $ext)\e[0m"
            disallowed_found=1
        fi
    done
    return $disallowed_found
}
# Check all staged files
staged_files=$(git diff --cached --name-only)
if [[ -z "$staged_files" ]]; then
    echo "No files staged for commit."
    exit 0
fi
echo "Scanning staged files for sensitive information and disallowed extensions..."
# Flags to track issues
secrets_found=0
disallowed_files_found=0
for file in $staged_files; do
    # Ensure the file exists (it might have been deleted in the commit)
    if [[ -f "$file" ]]; then
        echo "Checking file: $file"
        scan_for_secrets "$file"
        if [[ $? -ne 0 ]]; then
            secrets_found=1
        fi
        check_disallowed_files "$file"
        if [[ $? -ne 0 ]]; then
            disallowed_files_found=1
        fi
    fi
done
# Abort commit if any issues are detected
if [[ $secrets_found -ne 0 || $disallowed_files_found -ne 0 ]]; then
    echo -e "\e[31mCommit aborted due to detected issues.\e[0m"
    echo -e "\e[31mPlease resolve the warnings listed above and try again.\e[0m"
    exit 1
fi
echo "No issues detected. Proceeding with commit."
exit 0
